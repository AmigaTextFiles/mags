; graphics.library öffnen
;------------------------
 lea      gfxname(pc),a1       ;Name der Library
 moveq    #0,d0                ;Version der Library
 move.l   4.w,a6               ;Basisadresse der exec.library
 jsr      -552(a6)             ;OpenLibrary()
 tst.l    d0                   ;Lib geöffnet ?
 beq      ende                 ;nein -> ende
 lea      _GfxBase(pc),a0      ;Basisadresse der graphics.library
 move.l   d0,(a0)              ;in _GfxBase sichern
 
 
; Interrupts verbieten (Mulitasking abschalten)
;-----------------------------------------------
 jsr      -120(a6)             ;Disable() - Wir schalten hier das
                               ;Multitasking ab. Unser Programm hat
                               ;nun den ganzen Amiga für sich alleine.
 
 
; alte Copperliste speichern
;---------------------------
 move.l   _GfxBase(pc),a6
 move.l   $26(a6),OldCop       ;Auf Adresse $26 zur Basisadresse
                               ;der graphics.library liegt die Adresse
                               ;der aktuellen Copperliste.
 
 
; unsere eigende Copperliste starten
;------------------------------------
 move.l   #Copperliste,$dff080 ;Im Register COP1LC steht nun die Adresse
                               ;unserer Copperliste.
 move.w   d0,$dff088           ;Wir starten unsere Copperliste indem wir
                               ;irgendeinen Wert ins Register COPJMP1
                               ;reinschreiben, z.B. D0.
 
 
; pruefen ob wir schon an Zeile 255 angekommen sind
;---------------------------------------------------
Pruefe:
 cmpi.b   #$ff,$dff006         ;cmpi - compare immediate -
                               ;Inhahlt von Register VHPOSR ($dff006)
                               ;mit der Konstante $ff (255) vergleichen.
 bne.s    Pruefe               ;Vergleich nicht wahr -> Pruefe
 
 
; Das graue Stueck vom roten Balken bewegen
;-------------------------------------------
 bsr.s    BewegeRot            ;jede 1/50 Sekunde ausführen
 
 
; Das graue Stueck vom gruenen Balken bewegen
;---------------------------------------------
 bsr.w    BewegeGruen          ;jede 1/50 Sekunde ausführen
                       
; pruefen on wir noch auf Zeile 255 sind, und auf Zeile 0 warten
;----------------------------------------------------------------
Warte:
 cmpi.b   #$ff,$dff006         ;255 in VHPOSR ?
 beq.s    Warte                ;nein -> Warte
 
 
; linke Maustaste gedrueckt ?
;-----------------------------
 btst     #6,$bfe001           ;btst - test bit -
                               ;Testen ob Bit 6 im Register PRA ($bfe001)
                               ;gesetzt (also 1) ist.
 bne.s    Pruefe               ;nein -> Pruefe
 
 
; alte COPPERLIST wieder starten
;--------------------------------
 move.l  OldCop(pc),$dff080    ;In Register COP1LC steht nun wieder die
                               ;Adresse der alten Copperliste
 move.w  d0,$dff088            ;In COPJMP1 irgendeinen Wert -> alte
                               ;Copperliste starten
 
 
 move.l  4.w,a6                ;Basisadresse der exec.library
 jsr     -126(a6)              ;Enable() - So, nun können wieder mehrere
                               ;Programme den Amigs nutzen.
 
 
; Interrupts wieder zulassen (Multitasking einschalten)
;------------------------------------------------------
 move.l  4.w,a6                ;Basisadresse der exec.library
 jsr     -126(a6)              ;Enable() - So, nun können wieder mehrere
                               ;Programme den Amigs nutzen.
 
 
; graphics.library wieder schliessen
;------------------------------------
 move.l   _GfxBase(pc),a1      ;Basisadresse der graphics.library
 jsr      -414(a6)             ;CloseLibrary()
 
 
; Programm verlassen
;--------------------
ende
 moveq    #0,d0                ;Rückgabewert des Programmes
 rts                           ;und tschüß ...
 
 
; Datenbereich
;-------------- gfxname:  dc.b  'graphics.library',0
          cnop  0,2
_GfxBase: dc.l  0
OldCop:   dc.l  0
Links_r:  dc.w  0
Rechts_r: dc.w  0
Links_g:  dc.w  0
Rechts_g: dc.w  0
 
 
;Routine um das graue Stueck vom roten Balken zu bewegen
;--------------------------------------------------------
BewegeRot:
 cmpi.w   #78,Rechts_r         ;Schon 78 Zyklen (1/50 Sekunde) nach rechts
                               ;bewegt ?
 bne.s    NachRechts_r         ;nein -> NachRechts_r
 cmpi.w   #78,Links_r          ;Schon 78 Zyklen (1/50 Sekunde) nach links
                               ;bewegt ?
 bne.s    NachLinks_r          ;nein -> NachLinks_r
 clr.w    Rechts_r             ;"Rechts_r" auf 0 setzen
 clr.w    Links_r              ;"Links_r" auf 0 setzen
 rts
NachRechts_r:
 lea      Wait_rot+1,a0        ;1 zur effektiven Adresse von Wait_rot
                               ;addieren und nach A0 laden (ohne eine 1
                               ;zu addieren würden wir die X-Position
                               ;nach A0 laden, wir wollen aber die Y-Pos.)
 move.w   #17,d0               ;17 nach D0 kopieren (Zähler für Schleife)
loop_rot_rechts:
 addq.b   #2,(a0)              ;2 zum linken WAIT des grauen Stücks add.
 addq.b   #2,8(a0)             ;2 zum rechten WAIT des grauen Stücks add.
 add.w    #24,a0
 dbra     d0,loop_rot_rechts   ;Mit "dbra" können wir eine Zählschleife in
                               ;Assembler realisieren. Der Inhalt von D0
                               ;bei jedem Durchlauf runtergezählt. Ist -1
                               ;in D0 wird die Schleife beendet. Hier geht
                               ;die Schleife von "loop_rot_rechts" bis zum
                               ;"dbra"-Befehl und wir 18 mal durchlaufen.
                               ;18 mal deshalb, weil der rote Balken 18
                               ;Zeilen hoch ist und wir die WAITs in jeder
                               ;Zeile bewegen müssen.
 add.w    #1,Rechts_r          ;1 zu "Rechts_r" addieren
 rts
NachLinks_r:
 lea      Wait_rot+1,a0        ;Diese Routine funktioniert genauso wie
 move.w   #17,d0               ;beim nach rechts bewegen, nur eben ein
loop_rot_links:                ;"sub" statt ein "add" bei den WAITs.
 subq.b   #2,(a0)              ;                 |  |
 subq.b   #2,8(a0)             ;                 |  |
 add.w    #24,a0               ;                 |  |
 dbra     d0,loop_rot_links    ;                   /
 add.w    #1,Links_r           ;                  /
 rts
 
 
;Routine um das graue Stueck vom roten Balken zu bewegen
;--------------------------------------------------------
BewegeGruen:
 cmpi.w   #78,Links_g          ;Diese Routine funktioniert genauso wie
 bne.s    NachLinks_g          ;die für den roten Balken.
 cmpi.w   #78,Rechts_g         ;                 |  |
 bne.s    NachRechts_g         ;                 |  |
 clr.w    Links_g              ;                 |  |
 clr.w    Rechts_g             ;                 |  |
 rts                           ;                 |  |
NachRechts_g:                  ;                 |  |
 lea      Wait_gruen+1,a0      ;                 |  |
 move.w   #17,d0               ;                 |  |
loop_gruen_rechts:             ;                 |  |
 addq.b   #2,(a0)              ;                 |  |
 addq.b   #2,8(a0)             ;                 |  |
 add.w    #24,a0               ;                 |  |
 dbra     d0,loop_gruen_rechts ;                 |  |
 add.w    #1,Rechts_g          ;                 |  |
 rts                           ;                 |  |
NachLinks_g:                   ;                 |  |
 lea      Wait_gruen+1,a0      ;                 |  |
 move.w   #17,d0               ;                 |  |
loop_gruen_links:              ;                 |  |
 subq.b   #2,(a0)              ;                 |  |
 subq.b   #2,8(a0)             ;                 |  |
 add.w    #24,a0               ;                 |  |
 dbra     d0,loop_gruen_links  ;                   /
 add.w    #1,Links_g           ;                  /
 rts
