Da das File letztes Mal nicht der NoCover beilag habe ich mich entschlossen den Quelltext zu Veröffentlichen. Wer das klasse Tool haben will, muss man es in einen Editor einhacken, speichern und danach im CLI protect c:EA rweds eingeben. Das wars.

.key assign

assign <assign> exists >env:e

if "<assign>" eq ""
echo "Kein Assign angegeben"
skip ende
endif

if "$e" eq "<assign> not assigned"
skip neu
else
requestchoice "Easy Assign Nachricht" "Assign existiert bereits" "entfernen" "ändern" "hinzufügen" "lassen" >env:n
skip $n
endif

lab 1
assign <assign> remove
skip ende

lab 2
skip neu

lab 3 requestfile drawer=sys: title = "Schublade auswählen" drawersonly >env:a
if warn
skip ende
endif
assign <assign> $a add
skip ende

lab neu
requestfile drawer=sys: title = "Schublade auswählen" drawersonly >env:a
if warn
skip ende
endif
assign <assign> $a
skip ende

lab 0
skip ende

lab ende
del env:e env:n env:a >nil:
